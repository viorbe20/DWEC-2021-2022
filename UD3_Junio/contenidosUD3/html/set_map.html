  • 1. Indica qué es un Set y crea uno partiendo del array [1, 2, 3, 4, 4]. Justifica su tamaño.

    • Colección que permite almacenar valores únicos de cualquier tipo, desde valores primitivos hasta referencias a objetos.
    • Se puede iterar sobre estos elementos en orden de inserción.
    • Los valores en Set solo pueden aparecer una vez.

    
    //Declaramos el array
    let aNumbers = [1,2,3,4,4];
    
    //Función que crea un Set a partir de un Array
    function createSet(array) {
    let mySet = new Set(array);
    return mySet;
    }
    
    //Llamada a la función
    console.log(createSet(aNumbers));
    
    
    
    
        
    
    //Comprobamos longitud
    function getSetSize(param){
    return param.size;
    }
    
    //Llamada a la función
    console.log(getSetSize(createSet(aNumbers)));

    • Array.pop(). Elimina y devuelve el último elemento del array.

    • Array.shift(). Elimina y devuelve el primer elemento del array.

    myArray.splice(0,1);
    console.log(myArray.slice(0,1));

    • Array.isNaN(start, end). Determina cuando el valor es NaN o no.

  • 11. Averigua mediante código si WeakSet o WeakMap son iterables. Puedes utilizar for..of.

    • WeakSet Iterable 
    

    //Comprobación WeakSet iterable
    let student1 = {name: 'Ana'};
    let student2 = {name: 'Laura'};
    let student3 = {name: 'Carlota'};

    let wsStudents = new Set();
    wsStudents.add(student1).add(student2).add(student3);

    function interateWeakSet(param){
      try {
        let wsEx11 = param;
        for (const element of wsEx11) {
          console.log(element);
        }
        return wsEx11; 
      } catch (e) {
        console.error("No es iterable.", e.message);
      }
     }

    console.log(interateWeakSet(wsStudents));
